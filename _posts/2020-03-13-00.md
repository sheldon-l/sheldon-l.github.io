- `components/Calendar.vue` - scripts

```js
// copy `Events` code: https://vuetifyjs.com/en/components/calendars/


// Add/Modify these code in script
import { db } from '@/main';

export default {
  data: () => ({
    today: new Date().toISOString().substring(0,10),
    focus: new Date().toISOString().substring(0,10),
    type: 'month',
    typeToLabel: {
    month: 'Month',
    week: 'Week',
    day: 'Day',
    '4day': '4 Days',
    },
    // db
    name: null,
    details: null,
    start: null,
    end: null,
    color: "#1976D2",

    currentlyEditing: null,
    selectedEvent: {},
    selectedElement: null,
    selectedOpen: false,
    events: [],
    dialog: false,
  }),
  mounted () {
    this.getEvents();
  },
  methods: {
    // Work with firebase
    async getEvents () {
      let snapshot = await db.collection('calendarEvent').get();
      let events = [];
      snapshot.forEach(doc => {
        let appData = doc.data()
        appData.id = doc.id;
        events.push(appData)
      });
      this.events = events
    },
    editEvent(event) {
      this.currentlyEditing = event.id;
    },
    async updateEvent(event) {
      await db
        .collection('calendarEvent')
        .doc(this.currentlyEditing)
        .update({details: event.details})
      this.selectedOpen = false
      this.currentlyEditing = null
    },
    async deleteEvent(event) {
      await db
        .collection('calendarEvent')
        .doc(event)
        .delete();
      this.selectedOpen = false
      this.getEvents()
    },
    async addEvent() {
      if (this.name && this.start && this.end) {
        await db.collection('calendarEvent').add({
          name: this.name,
          details: this.details,
          start: this.start,
          end: this.end,
          color: this.color,
        });
        this.getEvents();
        this.name = "";
        this.details = "";
        this.start = "";
        this.end = "";
        this.color = "";
      } else {
        alert('Name, start and end date are required')
      }
    },
    updateRange ({ start, end }) {
      this.start = start
      this.end = end
    },
  }
```
